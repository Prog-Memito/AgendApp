# Para instalar las dependencias usar

npm run instalar 

al ejecutar "rpm run instalar" instala todas las dependencias que necesita el proyecto
tanto como "npm install express cors oracledb dotenv" que instala las dependencias para un
API REST y la conexion nativa con Oracle como el "npm install" que son las dependencias de ionic


# para iniciar el servidor de prueba usar:

cd backend
node index.js

# Para iniciar el proyecto de prueba usar:

cd frontend
ionic serve
